  // Typed helper for extracting data from Supabase responses
  export function unwrap<T>({ data, error }: { data: T | null; error: any }): T {
    if (error) throw error;
    if (data === null) throw new Error('Not found');
    return data;
  }

  import { useQuery } from '@tanstack/react-query';
  import { queryKeys } from '@/lib/queryKeys';

  // Fetch function — can be used in clientLoader or useQuery
  export async function fetchTown(townId: string) {
    return supabase
      .from('town')
      .select('*')
      .eq('id', townId)
      .single()
      .then(unwrap);

TASK 4: Create typed mutation hooks
Create packages/web/src/lib/mutations/ directory.

Create packages/web/src/lib/mutations/meetings.ts:
  import { useMutation, useQueryClient } from '@tanstack/react-query';
  import { queryKeys } from '@/lib/queryKeys';
  import { supabase } from '@/lib/supabase/client';

  // Insert a new meeting
  export function useCreateMeeting() {
    const queryClient = useQueryClient();
    return useMutation({
      mutationFn: async (data: CreateMeetingInput) => {
        const { data: meeting, error } = await supabase
          .from('meeting')
          .insert({ ...data, id: crypto.randomUUID() })
          .select()
          .single();
        if (error) throw error;
        return meeting;
      },
      onSuccess: (meeting) => {
        // Invalidate the meetings list for this board
        queryClient.invalidateQueries({ queryKey: queryKeys.meetings(meeting.board_id) });
      },
    });
  }

  // Update an existing meeting
  export function useUpdateMeeting() {
    const queryClient = useQueryClient();
    return useMutation({
      mutationFn: async ({ id, ...data }: UpdateMeetingInput) => {
        const { data: meeting, error } = await supabase
          .from('meeting')
          .update(data)
          .eq('id', id)
          .select()
          .single();
        if (error) throw error;
        return meeting;
      },
      onSuccess: (meeting) => {
        queryClient.invalidateQueries({ queryKey: queryKeys.meeting(meeting.id) });
        queryClient.invalidateQueries({ queryKey: queryKeys.meetings(meeting.board_id) });
      },
    });
  }

Create similar mutation files for: agenda items, board members, attendance, motions, votes.
All mutations follow this pattern:
  1. mutationFn: call supabase.from().insert/update/delete()
  2. onSuccess: call queryClient.invalidateQueries() for affected query keys

For the live meeting manager, add optimistic updates where appropriate:
  useMutation({
    mutationFn: ...,
    onMutate: async (variables) => {
      // Optimistically update the cache
      await queryClient.cancelQueries({ queryKey: queryKeys.meeting(meetingId) });
      const previous = queryClient.getQueryData(queryKeys.meeting(meetingId));
      queryClient.setQueryData(queryKeys.meeting(meetingId), (old) => ({
        ...old,
        // apply optimistic update
      }));
      return { previous };
    },
    onError: (err, variables, context) => {
      // Rollback on error
      queryClient.setQueryData(queryKeys.meeting(meetingId), context?.previous);
    },
    onSettled: () => {
      queryClient.invalidateQueries({ queryKey: queryKeys.meeting(meetingId) });
    },
  });

TASK 5: Create useRealtimeSubscription hook
Create packages/web/src/hooks/useRealtimeSubscription.ts:

  import { useEffect, useRef } from 'react';
  import { useQueryClient } from '@tanstack/react-query';
  import type { RealtimePostgresChangesPayload } from '@supabase/supabase-js';
  import { supabase } from '@/lib/supabase/client';

  interface UseRealtimeSubscriptionOptions {
    table: string;
    filter?: string;
    event?: 'INSERT' | 'UPDATE' | 'DELETE' | '*';
    onEvent: (payload: RealtimePostgresChangesPayload<Record<string, any>>) => void;
    enabled?: boolean;
  }

  /**
   * Subscribe to Supabase Realtime postgres_changes for a table.
   * Cleans up the subscription on unmount or when options change.
   * Used primarily in the live meeting manager for multi-device sync.
   */
  export function useRealtimeSubscription({
    table,
    filter,
    event = '*',
    onEvent,
    enabled = true,
  }: UseRealtimeSubscriptionOptions) {
    const onEventRef = useRef(onEvent);
    onEventRef.current = onEvent;

    useEffect(() => {
      if (!enabled) return;

      const channelName = filter ? `${table}:${filter}` : table;
      const channel = supabase
        .channel(channelName)
        .on(
          'postgres_changes',
          {
            event,
            schema: 'public',
            table,
            ...(filter ? { filter } : {}),
          },
          (payload) => onEventRef.current(payload)
        )
        .subscribe();

      return () => {
        supabase.removeChannel(channel);
      };
    }, [table, filter, event, enabled]);
  }

Usage example in a live meeting component:
  function LiveMeetingPage({ meetingId }: { meetingId: string }) {
    const queryClient = useQueryClient();

    // Sync agenda item changes across devices
    useRealtimeSubscription({
      table: 'agenda_item',
      filter: `meeting_id=eq.${meetingId}`,
      onEvent: () => {
        queryClient.invalidateQueries({ queryKey: queryKeys.meeting(meetingId) });
      },
    });

    // Sync vote records across devices
    useRealtimeSubscription({
      table: 'vote_record',
      filter: `meeting_id=eq.${meetingId}`,
      onEvent: () => {
        queryClient.invalidateQueries({ queryKey: queryKeys.meetingVotes(meetingId) });
      },
    });
  }

TASK 6: Create ConnectionStatusBar component
Create packages/web/src/components/ConnectionStatusBar.tsx:

This component shows the Supabase Realtime connection state. It replaces the old SyncStatusBar
that showed PowerSync sync queue status.

  import { useState, useEffect } from 'react';
  import { supabase } from '@/lib/supabase/client';
  import { Badge } from '@/components/ui/badge';
  import { ErrorBoundary } from 'react-error-boundary';

  type ConnectionState = 'connected' | 'connecting' | 'disconnected';

  function ConnectionStatusBarInner() {
    const [state, setState] = useState<ConnectionState>('connecting');

    useEffect(() => {
      // Subscribe to the Supabase Realtime connection state
      const channel = supabase.channel('__connection_status__').subscribe((status) => {
        if (status === 'SUBSCRIBED') setState('connected');
        else if (status === 'CLOSED' || status === 'CHANNEL_ERROR') setState('disconnected');
        else setState('connecting');
      });

      return () => { supabase.removeChannel(channel); };
    }, []);

    // Don't render anything when connected — no news is good news
    if (state === 'connected') return null;

    return (
      <div className="flex items-center gap-2 px-3 py-1 text-sm">
        {state === 'connecting' && (
          <Badge variant="outline" className="gap-1 border-amber-500 text-amber-700">
            <span className="h-2 w-2 rounded-full bg-amber-500 animate-pulse" />
            Connecting...
          </Badge>
        )}
        {state === 'disconnected' && (
          <Badge variant="outline" className="gap-1 border-red-500 text-red-700">
            <span className="h-2 w-2 rounded-full bg-red-500" />
            Disconnected — meeting data may be outdated
          </Badge>
        )}
      </div>
    );
  }

  // Wrap in ErrorBoundary — this component must never crash the meeting view
  export function ConnectionStatusBar() {
    return (
      <ErrorBoundary fallback={null}>
        <ConnectionStatusBarInner />
      </ErrorBoundary>
    );
  }

TASK 7: Update useCurrentUser hook
Update packages/web/src/hooks/useCurrentUser.ts to use React Query for the user's data,
so it benefits from caching and automatic refetch when the session refreshes.

  import { useQuery } from '@tanstack/react-query';
  import { queryKeys } from '@/lib/queryKeys';
  import { supabase } from '@/lib/supabase/client';

  export interface CurrentUser {
    id: string;              // Supabase auth user ID
    personId: string;        // PERSON entity ID
    email: string;
    townId: string | null;   // null if no town set up yet
    role: 'sys_admin' | 'admin' | 'staff' | 'board_member';
    govTitle: string | null;
    permissions: Record<string, boolean>;
  }

  async function fetchCurrentUser(): Promise<CurrentUser | null> {
    const { data: { session } } = await supabase.auth.getSession();
    if (!session) return null;

    // Extract custom claims from JWT payload
    const payload = JSON.parse(atob(session.access_token.split('.')[1]));
    const appMetadata = session.user.app_metadata;

    // Note: Supabase sets payload.role = "authenticated" (not our app role)
    // Our app role is in app_metadata.role (set by auth hook in session 01.08)
    const VALID_ROLES = ['sys_admin', 'admin', 'staff', 'board_member'] as const;
    const role = VALID_ROLES.includes(payload.role)
      ? payload.role
      : appMetadata.role ?? 'board_member';

    return {
      id: session.user.id,
      personId: payload.person_id ?? appMetadata.person_id,
      email: session.user.email!,
      townId: payload.town_id ?? appMetadata.town_id ?? null,
      role,
      govTitle: payload.gov_title ?? appMetadata.gov_title ?? null,
      permissions: payload.permissions ?? appMetadata.permissions ?? {},
    };
  }

  export function useCurrentUser() {
    return useQuery({
      queryKey: queryKeys.currentUser(),
      queryFn: fetchCurrentUser,
      staleTime: 5 * 60 * 1000, // JWT refresh happens every 1 hour; 5 min stale is fine
    });
  }
